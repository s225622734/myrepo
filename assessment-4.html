        main="S. thermophilus Base Composition", ylab="Count")

# Translate to proteins

prot_sth   <- lapply(cds_sth,   translate)

# Amino acid frequencies

barplot(as.numeric(aafreq_sth), names.arg=names(aafreq_sth),
